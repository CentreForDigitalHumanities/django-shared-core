        super(ApiError, self).__init__(*args)
        self.status_code = status_code
        self.message = message
        logger.error(f"Status {status_code}: {message}")



    def __str__(self):
        return 'ApiError ({}): {}'.format(self.status_code, self.message)

    def __repr__(self):
        return "<ApiError ({}): '{}'>".format(self.status_code, self.message)
